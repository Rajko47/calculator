\section*{Komptonovo raspršenje -- matematičke relacije}

1. \textbf{Očuvanje četvorovektora impulsa:}
\[
p^\mu + k^\mu = p'^\mu + k'^\mu,
\]
uz uslove
\[
p^2 = p'^2 = m^2, \qquad k^2 = k'^2 = 0.
\]

\bigskip

2. \textbf{Izlazna frekvencija fotona:}
\[
\omega'=\frac{\omega}{1+\dfrac{\hbar\omega}{mc^2}(1-\cos\theta)}.
\]

\bigskip

3. \textbf{Komptonova promena talasne dužine:}
\[
\lambda' - \lambda = \frac{h}{mc}(1-\cos\theta).
\]

\bigskip

4. \textbf{Klasični radijus elektrona:}
\[
r_0=\frac{e^2}{4\pi\varepsilon_0 mc^2}.
\]

\bigskip

5. \textbf{Klein--Nishina formula:}
\[
\frac{d\sigma}{d\Omega}
= \frac{r_0^2}{2}\left(\frac{\omega'}{\omega}\right)^2
\left(\frac{\omega}{\omega'}+\frac{\omega'}{\omega}-\sin^2\theta\right).
\]

\bigskip

6. \textbf{Thomsonov limit:}
\[
\frac{d\sigma}{d\Omega}\Big|_{\text{Thomson}}
= \frac{r_0^2}{2}(1+\cos^2\theta).
\]

\bigskip

7. \textbf{Ukupan presek (Klein--Nishina):}
\[
\begin{aligned}
\sigma(x)=2\pi r_0^2\Bigg[ &\frac{1+x}{x^3}
\left\{\frac{2x(1+x)}{1+2x}-\ln(1+2x)\right\} \\
&+\frac{\ln(1+2x)}{2x}-\frac{1+3x}{(1+2x)^2}\Bigg],
\end{aligned}
\]
gde je $x=\dfrac{\hbar\omega}{mc^2}$.

\bigskip

8. \textbf{Mandelstamovi invarianti:}
\[
s=(p+k)^2,\qquad t=(k-k')^2,\qquad u=(p-k')^2,
\]
uz relaciju
\[
s+t+u=2m^2.
\]

\bigskip

9. \textbf{QED amplituda (tree-level):}
\[
\mathcal{M} = -e^2\,\bar u(p')\left[
\frac{\slashed{\varepsilon}'^{\!*}(\slashed{p}+\slashed{k}+m)\slashed{\varepsilon}}
{(p+k)^2-m^2}
+
\frac{\slashed{\varepsilon}(\slashed{p}-\slashed{k}'+m)\slashed{\varepsilon}'^{\!*}}
{(p-k')^2-m^2}
\right]u(p).
\]

\bigskip

10. \textbf{Spin-prosečan kvadrat amplitude:}
\[
\overline{|\mathcal M|^2}=\frac{1}{4}\sum_{\text{spins, pol.}} |\mathcal M|^2.
\]
